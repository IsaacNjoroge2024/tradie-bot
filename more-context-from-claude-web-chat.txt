```python
position_size = (account_value * risk_pct) / (entry - stop_loss)
```

```python
position = (account * risk_pct) / (atr * multiplier * point_value)
```

### Spring Boot webhook endpoint
```java
@RestController
@RequestMapping("/webhook")
public class TradingViewWebhook {
    
    @PostMapping("/signal")
    public ResponseEntity<?> receiveSignal(
            @RequestBody SignalPayload signal,
            @RequestHeader("X-Passphrase") String passphrase) {
        
        if (!validatePassphrase(passphrase)) {
            return ResponseEntity.status(401).build();
        }
        
        Contract contract = contractMapper.map(signal.getTicker());
        Order order = orderBuilder.fromSignal(signal);
        ibClient.placeOrder(nextOrderId++, contract, order);
        
        return ResponseEntity.ok().build();
    }
}
```
